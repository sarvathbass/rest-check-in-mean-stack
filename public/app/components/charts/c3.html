(function () {
    'use strict';
    angular.module('mara').controller('chartsC3Controller',chartsC3Controller);
    chartsC3Controller.$inject = ['$timeout'];
    function chartsC3Controller($timeout) {
        var vm = this;
        vm.c3js1 = c3.generate({
            bindto: '#c3js1',
            data: {
                columns: [
                    ['data1', 30, 200, 100, 400, 150, 250],
                    ['data2', 50, 20, 10, 40, 15, 25]
                ]
            }
        });
        $timeout(function () {
            vm.c3js1.load({
                columns: [
                    ['data1', 230, 190, 300, 500, 300, 400]
                ]
            });
        }, 2000);
        $timeout(function () {
            vm.c3js1.load({
                columns: [
                    ['data3', 130, 150, 200, 300, 200, 100]
                ]
            });
        }, 2500);
        $timeout(function () {
            vm.c3js1.unload({
                ids: 'data1'
            });
        }, 3000);
    }
})();
